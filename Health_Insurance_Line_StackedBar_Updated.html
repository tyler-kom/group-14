#Importing the dataset and APIs

import pandas as pd
import matplotlib.pyplot as plt

# Load the dataset
data = pd.read_csv(r"/Users/anastasia/Downloads/states (1).csv")

data_without_california = data.copy()
data_without_california.drop(4, inplace=True)

# Filter for states that implemented Medicaid expansion in 2016
expansion_states = data_without_california[data_without_california['State Medicaid Expansion (2016)'] == True]

# Select relevant columns for plotting
expansion_states = expansion_states[['State', 'Medicaid Enrollment (2013)', 'Medicaid Enrollment (2016)']]

# Set 'State' as the index for better plotting
expansion_states.set_index('State', inplace=True)

# Plot the data
plt.figure(figsize=(10, 6))
plt.plot(expansion_states.index, expansion_states['Medicaid Enrollment (2013)'], 
         label="Medicaid Enrollment (2013)", color="blue", marker="o")
plt.plot(expansion_states.index, expansion_states['Medicaid Enrollment (2016)'], 
         label="Medicaid Enrollment (2016)", color="red", marker="s")

# Add labels and title
plt.xlabel("State")
plt.ylabel("Medicaid Enrollment")
plt.title("Change in Medicaid Enrollment (2013 vs 2016) in Expansion States without California")
plt.legend()
plt.xticks(rotation=45)
plt.tight_layout()

# Display the plot
plt.show()

# Load the dataset
data = pd.read_csv(r"/Users/anastasia/Downloads/states (1).csv")

# Filter for states that did not implement Medicaid expansion in 2016
non_expansion_states = data[data['State Medicaid Expansion (2016)'] == False]

# Select relevant columns for plotting
non_expansion_states = non_expansion_states[['State', 'Medicaid Enrollment (2013)', 'Medicaid Enrollment (2016)']]

# Set 'State' as the index for better plotting
non_expansion_states.set_index('State', inplace=True)

# Create the stacked bar graph
plt.figure(figsize=(12, 8))
plt.bar(non_expansion_states.index, non_expansion_states['Medicaid Enrollment (2013)'], 
        label="Medicaid Enrollment (2013)", color="blue")
plt.bar(non_expansion_states.index, non_expansion_states['Medicaid Enrollment (2016)'] - non_expansion_states['Medicaid Enrollment (2013)'], 
        bottom=non_expansion_states['Medicaid Enrollment (2013)'], label="Medicaid Enrollment (2016)", color="orange")

# Add labels and title
plt.xlabel("State")
plt.ylabel("Medicaid Enrollment")
plt.title("Change in Medicaid Enrollment (2013 vs 2016) in Non-Implemeneted Medicaid Expansion States")
plt.legend()
plt.xticks(rotation=45)
plt.tight_layout()

# Display the plot
plt.show()
