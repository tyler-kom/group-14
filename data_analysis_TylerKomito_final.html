# perform an independent t-test
import scipy.stats as stats
from sklearn.impute import SimpleImputer
df['State Medicaid Expansion (2016)'] = df['State Medicaid Expansion (2016)'].apply(lambda x: True if str(x).strip() == 'True' else False)

# Impute missing values using the mean of the column
imputer = SimpleImputer(strategy='median')
df_without_california['Medicaid Enrollment Change (2013-2016)'] = imputer.fit_transform(df_without_california[['Medicaid Enrollment Change (2013-2016)']])

# Separate the data into two groups: states that implemented the expansion plan and those that did not
expanded_states = df_without_california[df_without_california['State Medicaid Expansion (2016)'] == True]['Medicaid Enrollment Change (2013-2016)']
non_expanded_states = df_without_california[df_without_california['State Medicaid Expansion (2016)'] == False]['Medicaid Enrollment Change (2013-2016)']

print(expanded_states)

# Perform an independent t-test to compare the means of the two groups
t_statistic, p_value = stats.ttest_ind(expanded_states, non_expanded_states)

print(f"T-statistic: {t_statistic}")
print(f"P-value: {p_value}")

Conclusions: Even though the visualizations show that states that implemented the expansion plan typically experienced a greater increase in Medicaid enrollment, the independent t-test does not indicate that there is a statistically significant difference between the two sets.
